class Account {
public:
  void deposit(int n) {
    m_Mutex.lock();
    m_Balance += n;
    m_Mutex.unlock();
  }

  void withdraw(int n) {
    m_Mutex.lock();
    if (m_Balance >= n) {
      m_Balance -= n;
    }
    m_Mutex.unlock();
  }
};
\end{lstlisting}
\end{adjustbox}

   credit :: Integer -> Account -> STM ()
   credit amount account = do
       current <- readTVar account
       writeTVar account (current + amount)
 
   debit :: Integer -> Account -> STM ()
   debit amount account = do
       current <- readTVar account
       writeTVar account (current - amount)
